Pipes

Without using pipes today = {{ today }}

Using date pipes = {{ today | date }}

Using date pipes using Medium argument = {{ today | date : "medium" }}

Using date pipes using short = {{ today | date : "short" }}

Using date pipes using full date = {{ today | date : "fullDate" }}

Using long date pipes = {{ today | date : "longDate" }}

Using custom format = {{ today | date : "dd/mm/yyyy" }}
